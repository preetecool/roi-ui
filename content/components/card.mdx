
</InstallationTabsContent>

<InstallationTabsContent value="manual">

Copy and paste the TSX file:

    src="registry/brook/ui/card/card.tsx"
    title="components/ui/card.tsx"
    collapsible={true}
/>

Copy and paste the CSS file:

<ComponentSource
    src="registry/brook/ui/card/card.module.css"
    title="components/ui/card.module.css"
    language="css"
    collapsible={true}
/>
