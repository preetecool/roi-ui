## Responsive Charts

### Installation

```package-install
@visx/responsive
```

### Usage

<CodeBlock
  filename="responsive-chart.tsx"
  language="tsx"
  code={`import { ParentSize } from '@visx/responsive';
import { LineChart } from '@/registry/brook/ui/charts/line-chart';

export default function ResponsiveChart() {
  const data = [
    { date: '2023-01-01', value: 100, type: 'A' },
    { date: '2023-02-01', value: 150, type: 'A' },
    { date: '2023-03-01', value: 120, type: 'A' },
  ];

  return (
    <div style={{ width: '100%', height: '400px' }}>
      <ParentSize>
        {({ width, height }) => (
          <LineChart
            data={data}
            width={width}
            height={height}
          />
        )}
      </ParentSize>
    </div>
  );
}`}
/>
