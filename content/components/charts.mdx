## Line Chart

<ComponentPreview name="line-chart-demo" />

<h3 className="installation-heading">Installation</h3>

```package-install
npx shadcn@latest add https://roiui.com/r/line-chart.json
```

### Props

<PropTable data={[
  {
    prop: "data",
    type: "LineChartData[]",
    description: "Array of data points with date, value, and type properties"
  },
  {
    prop: "showXAxis",
    type: "boolean",
    default: "true",
    description: "Whether to display the X axis"
  },
  {
    prop: "showYAxis",
    type: "boolean",
    default: "true",
    description: "Whether to display the Y axis"
  },
  {
    prop: "showXGrid",
    type: "boolean",
    default: "true",
    description: "Whether to display vertical grid lines"
  },
  {
    prop: "showYGrid",
    type: "boolean",
    default: "true",
    description: "Whether to display horizontal grid lines"
  },
  {
    prop: "curve",
    type: '"linear" | "monotoneX" | "cardinal" | "basis" | "step"',
    default: '"linear"',
    description: "Type of curve interpolation for the line"
  },
  {
    prop: "showPoints",
    type: "boolean",
    default: "false",
    description: "Whether to display data points on the line"
  },
  {
    prop: "pointSize",
    type: "number",
    default: "4",
    description: "Size of the data points when showPoints is true"
  },
  {
    prop: "xAxisFormatter",
    type: "(value: any) => string",
    description: "Function to format X axis tick labels"
  },
  {
    prop: "animated",
    type: "boolean",
    default: "false",
    description: "Whether to animate the chart on load"
  }
]} />
