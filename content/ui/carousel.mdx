## Anatomy

<CodeBlock
filename="anatomy"
  language="tsx"
  code={`import { Carousel } from "@/components/ui/carousel";

<Carousel itemsPerView={3.2} gap={16}>
  <Carousel.Viewport>
    <Carousel.Content>
      <Carousel.Item index={0}>
        <div>Item 1</div>
      </Carousel.Item>
      <Carousel.Item index={1}>
        <div>Item 2</div>
      </Carousel.Item>
      <Carousel.Item index={2}>
        <div>Item 3</div>
      </Carousel.Item>
    </Carousel.Content>
  </Carousel.Viewport>
  <Carousel.Navigation />
  <Carousel.SRInfo />
  <Carousel.KeyboardHandler />
</Carousel>
`} />
